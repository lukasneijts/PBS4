9// Compute all forces on particles
10// This function returns the total potential energy of the system.
11double calculate_forces(struct Parameters *p_parameters, struct Colllist *p_colllist, struct Vectors *p_vectors)
12{
13 double Epot = 0.0;
14 struct Vec3D *f = p_vectors->f;
15 struct Vec3D *T = p_vectors->T;
16 struct Vec3D g;
17 g.x = p_parameters->g.x;
18 g.y = p_parameters->g.y;
19 g.z = p_parameters->g.z;
20 double * mass = p_vectors->mass;
21 // initialize the forces an torques with the gravitational force
22 const size_t num_part = p_parameters->num_part;
23 for (size_t i = 0; i < num_part; i++)
24 {
25 f[i].x = mass[i]*g.x; /*initialize forces to gravitational force*/
26 f[i].y = mass[i]*g.y;
27 f[i].z = mass[i]*g.z;
28 T[i] = (struct Vec3D){0.0, 0.0, 0.0};
29 }
30 Epot += calculate_forces_pp(p_parameters, p_colllist, p_vectors);
31 Epot += calculate_forces_pw(p_parameters, p_colllist, p_vectors);
32 return Epot;
33}

35// Compute all forces on particles die to particle-particle contacts
36// The function implement a soft-sphere model and used a collision list
37// This function returns the potential energy of (the concervative part of) these interactions
38double calculate_forces_pp(struct Parameters *p_parameters, struct Colllist *p_colllist, struct Vectors *p_vectors)
39{
40 double Epot = 0.0;
41 const double k_n_pp = p_parameters->k_n_pp;
42 const double eta_n_pp = p_parameters->eta_n_pp;
43 const double k_t_pp = p_parameters->k_t_pp;
44 const double eta_t_pp = p_parameters->eta_t_pp;
45 const double fric_pp = p_parameters->fric_pp;
46 struct Pair *nbr = p_colllist->nbr;
47 struct DeltaR *tijs = p_colllist->tij;
48 const size_t num_nbrs = p_colllist->num_nbrs;
49 const size_t num_part = p_parameters->num_part;
50 double * R = p_vectors->radius;
51 struct Vec3D *f = p_vectors->f;
52 struct Vec3D *T = p_vectors->T;
53 struct Vec3D *v = p_vectors->v;
54 struct Vec3D *omega = p_vectors->omega;
55 double * mass = p_vectors->mass;
56 double inv_mass_ref = 1.0/p_parameters->mass_ref;
57 for (size_t k = 0; k < num_nbrs; k++)
58 {
59 // for each pair in the neighbor list compute the pair forces
60 struct DeltaR rij = nbr[k].rij;
61 size_t i = nbr[k].i;
62 size_t j = nbr[k].j;
63 double mass_factor = sqrt(2.0*mass[i]*mass[j]/(mass[i]+mass[j])*inv_mass_ref);
64 // normal spring force
65 double r = sqrt(rij.sq);
66 double overlap = R[i]+R[j] - r;
67 double fr = k_n_pp * overlap / r;
68 struct DeltaR dfn;
69 dfn.x = fr * rij.x;
70 dfn.y = fr * rij.y;
71 dfn.z = fr * rij.z;
72 Epot += 0.5 * k_n_pp * overlap * overlap;
73 // normal dashpot force
74 struct Vec3D vij, vijn;
75 vij.x = v[i].x - v[j].x;
76 vij.y = v[i].y - v[j].y;
77 vij.z = v[i].z - v[j].z;
78 double fctr = (vij.x * rij.x + vij.y * rij.y + vij.z * rij.z) / rij.sq;
79 vijn.x = fctr * rij.x;
80 vijn.y = fctr * rij.y;
81 vijn.z = fctr * rij.z;
82 fr = -mass_factor*eta_n_pp;
83 dfn.x += fr * vijn.x;
84 dfn.y += fr * vijn.y;
85 dfn.z += fr * vijn.z;
86 // tangential spring force
87 struct DeltaR tij = tijs[k];
88 fr = -k_t_pp;
89 struct DeltaR dft;
90 dft.x = fr * tij.x;
91 dft.y = fr * tij.y;
92 dft.z = fr * tij.z;
93 // tangential dashpot force
94 struct Vec3D vijt;
95 vijt.x = vij.x - vijn.x;
96 vijt.y = vij.y - vijn.y;
97 vijt.z = vij.z - vijn.z;
98 vijt.x -= 0.5 * ((omega[i].y + omega[j].y) * rij.z -
99 (omega[i].z + omega[j].z) * rij.y);
100 vijt.y -= 0.5 * ((omega[i].z + omega[j].z) * rij.x -
101 (omega[i].x + omega[j].x) * rij.z);
102 vijt.z -= 0.5 * ((omega[i].x + omega[j].x) * rij.y -
103 (omega[i].y + omega[j].y) * rij.x);
104 fr = -mass_factor*eta_t_pp;
105 dft.x += fr * vijt.x;
106 dft.y += fr * vijt.y;
107 dft.z += fr * vijt.z;
108
109 //If tangential force is too large then sliding takes place
110 dfn.sq = dfn.x * dfn.x + dfn.y * dfn.y + dfn.z * dfn.z;
111 dft.sq = dft.x * dft.x + dft.y * dft.y + dft.z * dft.z;
112 if (dft.sq >= fric_pp * fric_pp * dfn.sq) //sliding
113 {
114 fr = fric_pp * sqrt(dfn.sq / dft.sq);
115 dft.x *= fr;
116 dft.y *= fr;
117 dft.z *= fr;
118 // when sliding set the tangential displacement such that the sticking force (nearly) equals the sliding force
119 tijs[k].x = -dft.x / k_t_pp;
120 tijs[k].y = -dft.y / k_t_pp;
121 tijs[k].z = -dft.z / k_t_pp;
122 }
123 else
124 Epot += 0.5 * k_t_pp * tij.sq;
125 struct Vec3D df;
126 df.x = dfn.x + dft.x;
127 df.y = dfn.y + dft.y;
128 df.z = dfn.z + dft.z;
129 struct Vec3D dT;
130 dT.x = 0.5 * (rij.z * dft.y - rij.y * dft.z);
131 dT.y = 0.5 * (rij.x * dft.z - rij.z * dft.x);
132 dT.z = 0.5 * (rij.y * dft.x - rij.x * dft.y);
133 f[i].x += df.x;
134 f[i].y += df.y;
135 f[i].z += df.z;
136 f[j].x -= df.x;
137 f[j].y -= df.y;
138 f[j].z -= df.z;
139 T[i].x += dT.x;
140 T[i].y += dT.y;
141 T[i].z += dT.z;
142 T[j].x += dT.x;
143 T[j].y += dT.y;
144 T[j].z += dT.z;
145 }
146 return Epot;
147}

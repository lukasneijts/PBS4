85int main(void)
86{
87 struct Vectors vectors;
88 struct Parameters parameters;
89 struct Nbrlist nbrlist;
90 struct Colllist colllist;
91 size_t step = 0;
92 double Ekin, Epot;
93
94 set_parameters(&parameters);
95 alloc_memory(&parameters, &vectors, &nbrlist, &colllist);
96 if(parameters.load_restart == 1)
97 {
98 load_restart(&parameters, &vectors);
99 }
100 else
101 initialise(&parameters, &vectors);
102
103 build_nbrlist(&parameters, &vectors, &nbrlist);
104 update_colllist(&parameters, &vectors, &nbrlist, &colllist);
105 Epot = calculate_forces(&parameters, &colllist, &vectors);
106 record_trajectories_xyz(1, &parameters, &vectors);
107
108 while (step < parameters.num_dt_steps) //start of the velocity-Verlet loop
109 {
110
111 step++;
112 vectors.time += parameters.dt;
113
114 Ekin = update_velocities_half_dt(&parameters, &nbrlist, &vectors);
115 update_positions(&parameters, &nbrlist, &vectors);
116
117 update_tangential_displacements(&parameters, &vectors, &colllist);
118 boundary_conditions(&parameters, &vectors);
119 update_nbrlist(&parameters, &vectors, &nbrlist);
120 update_colllist(&parameters, &vectors, &nbrlist, &colllist);
121 Epot = calculate_forces(&parameters, &colllist, &vectors);
122 Ekin = update_velocities_half_dt(&parameters, &nbrlist, &vectors);
123
124 if (step%parameters.num_dt_printf ==0) printf("Step %lu, Time %g, Z %g, Epot %g, Ekin %g, Etot %g\n", (long unsigned) step, vectors.time,
125 2.0*((double) colllist.num_nbrs)/((double) parameters.num_part),
126 Epot, Ekin, Epot+Ekin); //Z is the coordination number
127 if (step%parameters.num_dt_traj ==0) record_trajectories_xyz(0,&parameters,&vectors);
128 if (step%parameters.num_dt_restart == 0) save_restart(&parameters,&vectors);
129 }
130 save_restart(&parameters,&vectors);
131 free_memory(&vectors, &nbrlist, &colllist);
132
133 return 0;
134}

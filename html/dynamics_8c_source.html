7// This function updates particle positions using their velocities.
8// The positions are advanced by one full time step (dt), and displacement vectors
9// (dr) for one time step are updated. The displacement since the last neighbor
10// list creation (stored in p_nbrlist->dr) is also updated for each particle.
11void update_positions(struct Parameters *p_parameters, struct Nbrlist *p_nbrlist, struct Vectors *p_vectors)
12{
13 struct Vec3D dr_loc;
14 struct Vec3D *r = p_vectors->r; // Particle positions
15 struct Vec3D *dr = p_vectors->dr; // Displacement in one timestep
16 struct Vec3D *v = p_vectors->v; // Particle velocities
17 struct DeltaR *dr_nbrlist = p_nbrlist->dr; // Displacement since last neighbor list creation
18 size_t num_part = p_parameters->num_part;
19 double dt = p_parameters->dt;
20
21 // Loop over all particles to update their positions
22 for (size_t i = 0; i < num_part; i++)
23 {
24 dr_loc.x = v[i].x * dt; // Compute displacement in x-direction for one timestep
25 dr_loc.y = v[i].y * dt; // Compute displacement in y-direction for one timestep
26 dr_loc.z = v[i].z * dt; // Compute displacement in z-direction for one timestep
27
28 dr[i] = dr_loc; // Store the displacement for this timestep
29 r[i].x += dr_loc.x; // Update position in x-direction
30 r[i].y += dr_loc.y; // Update position in y-direction
31 r[i].z += dr_loc.z; // Update position in z-direction
32
33 // Update the displacement since last neighbor list creation
34 dr_nbrlist[i].x += dr_loc.x;
35 dr_nbrlist[i].y += dr_loc.y;
36 dr_nbrlist[i].z += dr_loc.z;
37 dr_nbrlist[i].sq = (dr_nbrlist[i].x) * (dr_nbrlist[i].x) +
38 (dr_nbrlist[i].y) * (dr_nbrlist[i].y) +
39 (dr_nbrlist[i].z) * (dr_nbrlist[i].z); // Square of total displacement since last neighbor list creation
40 }
41}

43void update_tangential_displacements(struct Parameters *p_parameters, struct Vectors *p_vectors, struct Colllist *p_colllist)
44{
45/* Upon collision, if it is sticking, a tangential displacement builds up.
46First the tangential displacement, vijt, is computed for pair ij.
47This velocity is used to update the tangential displacement: tij(t+dt) = tij(t) + vijt(t+0.5*dt)*dt */
48 size_t i,j,k;
49 double fctr, dt = p_parameters->dt;
50 const size_t num_nbrs = p_colllist->num_nbrs;
51 struct DeltaR rij;
52 struct Vec3D vij, vijn, vijt;
53 struct Vec3D *v, *omega, *vw;
54 struct Pair * nbr;
55 struct DeltaR *tij;
56 v = p_vectors->v;
57 omega = p_vectors->omega;
58 nbr = p_colllist->nbr;
59 tij = p_colllist->tij;
60 vw = p_colllist->vw;
61
62 for (k=0; k < num_nbrs; k++)
63 {
64 // for each pair in collision list determine relative tangential velocity also taking into account rotation
65 rij = nbr[k].rij;
66 i = nbr[k].i;
67 j = nbr[k].j;
68 vij.x = v[i].x - v[j].x;
69 vij.y = v[i].y - v[j].y;
70 vij.z = v[i].z - v[j].z;
71 fctr = (vij.x*rij.x+vij.y*rij.y+vij.z*rij.z)/rij.sq;
72 vijn.x = fctr*rij.x;
73 vijn.y = fctr*rij.y;
74 vijn.z = fctr*rij.z;
75 vijt.x = vij.x-vijn.x;
76 vijt.y = vij.y-vijn.y;
77 vijt.z = vij.z-vijn.z;
78 vijt.x -= 0.5*((omega[i].y+omega[j].y)*rij.z-
79 (omega[i].z+omega[j].z)*rij.y);
80 vijt.y -= 0.5*((omega[i].z+omega[j].z)*rij.x-
81 (omega[i].x+omega[j].x)*rij.z);
82 vijt.z -= 0.5*((omega[i].x+omega[j].x)*rij.y-
83 (omega[i].y+omega[j].y)*rij.x);
84 // for each pair in the collisionlist update relative tangential displacements
85 tij[k].x += vijt.x*dt;
86 tij[k].y += vijt.y*dt;
87 tij[k].z += vijt.z*dt;
88 tij[k].sq = tij[k].x *tij[k].x + tij[k].y*tij[k].y + tij[k].z*tij[k].z;
89 }
90 size_t num_w = p_colllist->num_w;
91 size_t * indcs_w = p_colllist->indcs_w;
92 struct DeltaR * riw = p_colllist->riw;
93 struct DeltaR * tiw = p_colllist->tiw;
94 for (j=0; j < num_w; ++j)
95 {
96 i = indcs_w[j];
97 rij = riw[j];
98 vij.x = v[i].x-vw[j].x;
99 vij.y = v[i].y-vw[j].y;
100 vij.z = v[i].z-vw[j].z;
101 fctr = (vij.x*rij.x+vij.y*rij.y+vij.z*rij.z)/rij.sq;
102 vijn.x = fctr*rij.x;
103 vijn.y = fctr*rij.y;
104 vijn.z = fctr*rij.z;
105 vijt.x = vij.x-vijn.x;
106 vijt.y = vij.y-vijn.y;
107 vijt.z = vij.z-vijn.z;
108 vijt.x -= (omega[i].y*rij.z-omega[i].z*rij.y);
109 vijt.y -= (omega[i].z*rij.x-omega[i].x*rij.z);
110 vijt.z -= (omega[i].x*rij.y-omega[i].y*rij.x);
111 tiw[j].x += vijt.x*dt;
112 tiw[j].y += vijt.y*dt;
113 tiw[j].z += vijt.z*dt;
114 tiw[j].sq = tiw[j].x *tiw[j].x + tiw[j].y*tiw[j].y + tiw[j].z*tiw[j].z;
115 }
116}

118// This function updates particle velocities by half a time step using the current forces.
119// The updated velocities are used in the velocity-Verlet integration scheme.
120// The function also calculates and returns the kinetic energy of the system.
121double update_velocities_half_dt(struct Parameters *p_parameters, struct Nbrlist *p_nbrlist, struct Vectors *p_vectors)
122{
123 double Ekin = 0.0, Ekin_rot = 0.0;
124 double *R = p_vectors->radius;
125 double *mass = p_vectors->mass;
126 const double factor = 0.5 * p_parameters->dt;
127 size_t num_part = p_parameters->num_part;
128 struct Vec3D *v, *omg, *f, *T;
129 v = p_vectors->v;
130 omg = p_vectors->omega;
131 f = p_vectors->f;
132 T = p_vectors->T;
133 for (size_t i = 0; i < p_parameters->num_part; i++)
134 {
135 double I = 0.4*mass[i]*R[i]*R[i];
136 v[i].x += factor * f[i].x/mass[i];
137 v[i].y += factor * f[i].y/mass[i];
138 v[i].z += factor * f[i].z/mass[i];
139 Ekin += mass[i]*(v[i].x * v[i].x + v[i].y * v[i].y + v[i].z * v[i].z);
140 omg[i].x += factor*T[i].x/I;
141 omg[i].y += factor*T[i].y/I;
142 omg[i].z += factor*T[i].z/I;
143 Ekin_rot += I*(omg[i].x*omg[i].x+omg[i].y*omg[i].y+omg[i].z*omg[i].z);
144 }
145 Ekin = 0.5* (Ekin + Ekin_rot);
146 return Ekin;
147}

149// This function applies periodic boundary conditions to ensure particles stay inside the simulation box.
150// If a particle moves beyond the box, it is wrapped around to the opposite side.
151void boundary_conditions(struct Parameters *p_parameters, struct Vectors *p_vectors)
152{
153 struct Vec3D invL; // Inverse of the box size
154 struct Vec3D *r = p_vectors->r; // Particle positions
155 struct Vec3D L = p_parameters->L; // Box dimensions
156 size_t num_part = p_parameters->num_part; // Number of particles
157
158 invL.x = 1.0 / L.x;
159 invL.y = 1.0 / L.y;
160 invL.z = 1.0 / L.z;
161
162 // Loop over all particles and apply periodic boundary conditions
163 for (size_t i = 0; i < num_part; i++)
164 {
165 r[i].x -= L.x * floor(r[i].x * invL.x); // Apply periodic boundary in x-direction
166 r[i].y -= L.y * floor(r[i].y * invL.y); // Apply periodic boundary in y-direction
167 r[i].z -= L.z * floor(r[i].z * invL.z); // Apply periodic boundary in z-direction
168 }

44void build_celllist(struct Parameters *p_parameters, struct Vectors *p_vectors, struct Celllist *p_celllist)
45/* Build the cell-linked-list */
46{
47 struct Index3D size_grid, indx;
48 size_t Mtot, icell;
49 const double rlist = p_parameters->r_cut + p_parameters->r_shell;
50 struct Vec3D mL;
51 const size_t num_part = p_parameters->num_part;
52 size_t *particle2cell, *head, *celllist;
53 struct Vec3D *r;
54
55 size_grid.i = floor(p_parameters->L.x / rlist);
56 size_grid.j = floor(p_parameters->L.y / rlist);
57 size_grid.k = floor(p_parameters->L.z / rlist);
58 p_celllist->size_grid.i = size_grid.i;
59 p_celllist->size_grid.j = size_grid.j;
60 p_celllist->size_grid.k = size_grid.k;
61 Mtot = size_grid.i * size_grid.j * size_grid.k;
62 if (Mtot > p_celllist->num_cells_max)
63 {
64 p_celllist->head = (size_t *)realloc(p_celllist->head, Mtot * sizeof(size_t));
65 p_celllist->num_cells_max = Mtot;
66 }
67 if (num_part > p_celllist->num_part_max)
68 {
69 p_celllist->particle2cell = (size_t *)realloc(p_celllist->particle2cell, num_part * sizeof(size_t));
70 p_celllist->list = (size_t *)realloc(p_celllist->list, num_part * sizeof(size_t));
71 p_celllist->num_part_max = num_part;
72 }
73 mL.x = ((double)size_grid.i) / p_parameters->L.x;
74 mL.y = ((double)size_grid.j) / p_parameters->L.y;
75 mL.z = ((double)size_grid.k) / p_parameters->L.z;
76 head = p_celllist->head;
77 for (icell = 0; icell < Mtot; ++icell)
78 head[icell] = SIZE_MAX;
79 particle2cell = p_celllist->particle2cell;
80 celllist = p_celllist->list;
81 r = p_vectors->r;
82 for (size_t i = (num_part - 1); i != SIZE_MAX; --i)
83 // Note that within a cell particles will be ordered in a descending way in the cell-linked list
84 {
85 indx.i = floor(r[i].x * mL.x);
86 indx.j = floor(r[i].y * mL.y);
87 indx.k = floor(r[i].z * mL.z);
88 icell = indx.i + size_grid.i * (indx.j + indx.k * size_grid.j);
89 particle2cell[i] = icell;
90 celllist[i] = head[icell];
91 head[icell] = i;
92 }

152 PairNBs are included to the neighbor list of their distance is less than r_cut+r_shell. */

266 }
267 else //swap particles i and j
268 {
269 nbr[num_nbrs].i = j;

271 rij.x = -rij.x;
272 rij.y = -rij.y;
273 rij.z = -rij.z;

277 num_nbrs++;

279 }
280 }
281 }

283 // The neighbor list is fully sorted. The entries are such that nbr.i < nbr.j. nbr.i is ascending and for fixed i, j is ascending.
284 size_t cnt_sum = 0;
285 for (size_t i = 0; i < num_part; ++i)
286 {
287 size_t tmp = nbr_cnt[i];
288 nbr_cnt[i] = cnt_sum;
289 cnt_sum += tmp;
290 }
291 p_nbrlist->nbr_tmp = (struct Pair *)realloc(p_nbrlist->nbr_tmp, num_nbrs_max * sizeof(struct Pair));
292 struct Pair *nbr_tmp = p_nbrlist->nbr_tmp;
293 for (size_t k = 0; k < num_nbrs; ++k)
294 {
295 size_t i = (nbr[k].i < nbr[k].j ? nbr[k].i : nbr[k].j);
296 nbr_tmp[nbr_cnt[i]++] = nbr[k];
297 }
298 p_nbrlist->nbr = nbr_tmp;
299 p_nbrlist->nbr_tmp = nbr;
300 nbr = p_nbrlist->nbr;
301 nbr_tmp = p_nbrlist->nbr_tmp;
302 size_t k = 0;
303 for (size_t i = 0; i < num_part; k = nbr_cnt[i], i++)

305 p_nbrlist->num_nbrs = num_nbrs;
306 p_nbrlist->dr = (struct DeltaR *)realloc(p_nbrlist->dr, num_part * sizeof(struct DeltaR));

308 p_nbrlist->dr[i] = dr;
309}

311int cmp_sort_nbr(const void *p1, const void *p2)
312{
313 const struct Pair *p_nbr1 = p1;
314 const struct Pair *p_nbr2 = p2;

321int update_nbrlist(struct Parameters *p_parameters, struct Vectors *p_vectors, struct Nbrlist *p_nbrlist)
322/* Update the connecting vectors of the neighbor list and if needed rebuild it.*/
323{
324 const double dr_sq_max = 0.25 * (p_parameters->r_shell * p_parameters->r_shell);
325 int isRebuild = 0;
326 struct DeltaR rij;
327 struct Pair *nbr = p_nbrlist->nbr;
328 struct Vec3D *dr = p_vectors->dr;
329 // The neighbor list needs to be rebuild if one of the particles has displaced more then 0.5*r_shell
330 for (size_t i = 0; i < p_parameters->num_part; ++i)
331 if ((p_nbrlist->dr[i].sq) > dr_sq_max)
332 {
333 isRebuild = 1;
334 }
335 if (isRebuild) // rebuild neighbor list
336 build_nbrlist(p_parameters, p_vectors, p_nbrlist);
337 else // If no rebuild is needed, update the values of the connecting vectors
338 {
339 for (size_t k = 0; k < p_nbrlist->num_nbrs; ++k)
340 {
341 size_t i = nbr[k].i;
342 size_t j = nbr[k].j;
343 rij = nbr[k].rij;
344 rij.x += (dr[i].x - dr[j].x);
345 rij.y += (dr[i].y - dr[j].y);
346 rij.z += (dr[i].z - dr[j].z);
347 rij.sq = rij.x * rij.x + rij.y * rij.y + rij.z * rij.z;
348 p_nbrlist->nbr[k].rij = rij;
349 }
350 }
351 return isRebuild;
352}
